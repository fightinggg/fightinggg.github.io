#include<bits/stdc++.h>
using namespace std;

int dfs(int*a,int n){
    if(n==3)return a[0]+a[1]>a[2];
    
    int ret=0,ret_array[12]{};
    int b[12],b_=0;
    
    for(int i=1;i<n;i++){
        for(int j=i+1;j<n;j++){
            b_=0;
            for(int k=1;k<n;k++){
                if(k==i||k==j)continue;
                b[b_++]=a[k];
            }
            int tmp=dfs(b,n-3);
            if(tmp+(a[0]+a[i]>a[j])>ret){
                ret=tmp+(a[0]+a[i]>a[j]);
                memcpy(ret_array,b,sizeof(int)*(n-3));
                ret_array[n-3]=a[0];
                ret_array[n-2]=a[i];
                ret_array[n-1]=a[j];
            }
        }
    }
    memcpy(a,ret_array,sizeof(int)*n);
    return ret;
}

int main(){
    int a[12],T;
    scanf("%d",&T);
    
    for(int times=1;times<=T;times++){
        for(int i=0;i<12;i++)scanf("%d",a+i);
        sort(a,a+12);
        int ans=dfs(a,12);
        printf("Case #%d: %d\n",times,ans);
        for(int i=0;i<12;i+=3){
            if(a[i]+a[i+1]>a[i+2]){
                printf("%d %d %d\n",a[i],a[i+1],a[i+2]);
            }
        }
    }
}
